## ğŸ” ×”×¡×‘×¨×™× ×•×“×•×’×××•×ª

### `add(item)`
```python
# Add item to the set
colors = {"red", "green"}
colors.add("blue")  # {'red', 'green', 'blue'}
````

### `remove(item)`

```python
# Remove item from set
colors = {"red", "green", "blue"}
colors.remove("green")  # {'red', 'blue'}
# KeyError if item doesn't exist
```

### `discard(item)`

```python
# Remove item if it exists (no error if not)
colors = {"red", "green"}
colors.discard("blue")  # No error even if 'blue' not found
```

### `pop()`

```python
# Remove and return a random item
numbers = {1, 2, 3}
item = numbers.pop()  # Removes random element
```

### `clear()`

```python
# Remove all items from the set
numbers = {1, 2, 3}
numbers.clear()  # set()
```

### `copy()`

```python
# Create a shallow copy of the set
a = {1, 2, 3}
b = a.copy()  # {1, 2, 3}
```

### `union(set2)` / `|`

```python
# Combine all unique items from both sets
a = {1, 2}
b = {2, 3}
print(a.union(b))  # {1, 2, 3}
print(a | b)       # Same result
```

### `intersection(set2)` / `&`

```python
# Items common to both sets
a = {1, 2, 3}
b = {2, 3, 4}
print(a.intersection(b))  # {2, 3}
print(a & b)              # Same result
```

### `difference(set2)` / `-`

```python
# Items in a but not in b
a = {1, 2, 3}
b = {2, 4}
print(a.difference(b))  # {1, 3}
print(a - b)            # Same result
```

### `symmetric_difference(set2)` / `^`

```python
# Items in either set, but not both
a = {1, 2, 3}
b = {2, 4}
print(a.symmetric_difference(b))  # {1, 3, 4}
print(a ^ b)                      # Same result
```

### `update(set2)`

```python
# Add all items from another set
a = {1, 2}
b = {2, 3}
a.update(b)  # a becomes {1, 2, 3}
```

### `intersection_update(set2)`

```python
# Keep only shared items
a = {1, 2, 3}
b = {2, 4}
a.intersection_update(b)  # a = {2}
```

### `difference_update(set2)`

```python
# Remove items found in another set
a = {1, 2, 3}
b = {2, 4}
a.difference_update(b)  # a = {1, 3}
```

### `symmetric_difference_update(set2)`

```python
# Keep only items that appear in one of the sets, not both
a = {1, 2, 3}
b = {2, 4}
a.symmetric_difference_update(b)  # a = {1, 3, 4}
```
