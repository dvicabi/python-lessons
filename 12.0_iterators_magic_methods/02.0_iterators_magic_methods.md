        value = self.current
        self.current += 1
        return value

class Doubler:
    def __init__(self, start, limit):
        self.value = start
        self.limit = limit

    def __iter__(self):
        return self

    def __next__(self):
        if self.value > self.limit:
            raise StopIteration
        current = self.value
        self.value *= 2
        return current

for num in Doubler(1, 20):
    print(num)  # 1, 2, 4, 8, 16
